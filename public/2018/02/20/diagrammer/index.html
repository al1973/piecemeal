DiagrammeR

2018/02/20 
library(DiagrammeR)
grViz("
  digraph example{
# node statements
    node [shape = oval,
          fontname = Helvetica,
          penwidth = 2.0,
          color = olive]
    A, B; C; D; E; F
    node [shape = box]
    1; 2; 3; 4; 5; 6; 7; 8
# edge statements
    edge[arrowhead = vee,
         color = sienna]
    A->1; B->2; B->3; B->4; C->A
    1->D; E->A; 2->4; 1->5; 1->F
    E->6; 4->6; 5->7; 6->7;
    3->8 [label = ' label!',
          fontname = Helvetica]
# graph statements
   graph [nodesep = 0.1]
  }")

Create the graph object, incorporating the NDFs and the EDFs

graph <-
        DiagrammeR::create_graph(
                nodes_df = all_nodes,
                edges_df = all_edges)

View the graph

render_graph(graph)
